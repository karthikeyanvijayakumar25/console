44 static inline unsigned int full_name_hash(const void *salt, const unsigned char * name, size_t len) {
45 
46  uint32_t hash = 0;
47  (void)salt;
48  while (len--) {
49  hash = (hash << 4) | (hash >> 28);
50  hash ^= *(name++);
51  }
52  return hash;
53 }
