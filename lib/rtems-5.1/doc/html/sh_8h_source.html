193 static inline uint32_t sh_swap_u32(
194  uint32_t value
195 )
196 {
197  uint32_t swapped;
198 
199  __asm__ volatile (
200  "swap.b %1,%0; "
201  "swap.w %0,%0; "
202  "swap.b %0,%0"
203  : "=r" (swapped)
204  : "r" (value) );
205 
206  return( swapped );
